function attachSearchDropdown_(rootSelect, searchFn) { const parent = rootSelect.parentNode; const doc = parent.ownerDocument ? parent.ownerDocument : { createElement() { return { style: {}, className: '', classList: { add() {} }, appendChild() {}, addEventListener() {} }; } }; const wrapper = doc.createElement('div'); wrapper.className = 'search-dropdown-host'; parent.insertBefore(wrapper, rootSelect); const valueDisplay = doc.createElement('div'); valueDisplay.className = 'search-dropdown-value'; const valueText = doc.createElement('span'); valueText.className = 'search-dropdown-value-text placeholder'; const arrow = doc.createElement('div'); arrow.className = 'search-dropdown-trigger'; valueDisplay.appendChild(valueText); valueDisplay.appendChild(arrow); wrapper.appendChild(valueDisplay); const panel = doc.createElement('div'); panel.className = 'search-dropdown-panel'; panel.style.display = 'none'; const input = doc.createElement('input'); input.type = 'text'; input.className = 'search-dropdown-input'; input.placeholder = 'Search...'; rootSelect.classList.add('search-dropdown-select'); rootSelect.size = 8; panel.appendChild(input); panel.appendChild(rootSelect); wrapper.appendChild(panel); const allOptions = Array.from(rootSelect.options).map(function(opt) { const ds = opt.dataset || {}; return { value: opt.value, label: opt.textContent, startDate: ds.start || null, endDate: ds.end || null }; }); let searchTimer = null; function cancelSearch_() { if (searchTimer !== null) { clearTimeout(searchTimer); searchTimer = null; } } function applyOptions_(options) { const prev = rootSelect.value || ''; rootSelect.innerHTML = ''; options.forEach(function(opt) { const el = doc.createElement('option'); el.value = opt.value; el.textContent = opt.label; rootSelect.appendChild(el); }); rootSelect.value = options.some(function(o) { return o.value === prev; }) ? prev : (options.length > 0 ? options[0].value : ''); } function refreshOptions_(query) { applyOptions_(searchFn(query, allOptions)); } function updateDisplay_() { const opt = rootSelect.options[rootSelect.selectedIndex]; if (opt && opt.value) { valueText.textContent = opt.textContent; valueText.className = 'search-dropdown-value-text'; } else { valueText.textContent = opt ? opt.textContent : ''; valueText.className = 'search-dropdown-value-text placeholder'; } } function isInWrapper_(target) { let node = target; while (node) { if (node === wrapper) return true; node = node.parentNode; } return false; } function hidePanel_() { panel.style.display = 'none'; input.value = ''; } function commitSelection_() { updateDisplay_(); hidePanel_(); } function openDropdown_() { cancelSearch_(); refreshOptions_(''); panel.style.display = 'block'; input.focus(); } refreshOptions_(''); updateDisplay_(); // Toggle on value display click; preventDefault keeps focus in panel valueDisplay.addEventListener('mousedown', function(event) { event.preventDefault(); if (panel.style.display === 'none') { openDropdown_(); } else { hidePanel_(); } }); // Close when clicking outside the wrapper if (doc.addEventListener) { doc.addEventListener('mousedown', function(event) { if (panel.style.display !== 'none' && !isInWrapper_(event.target)) { commitSelection_(); } }); } // Search input filtering with debounce input.addEventListener('input', function() { cancelSearch_(); const query = input.value || ''; if (!query.trim()) { refreshOptions_(''); return; } searchTimer = setTimeout(function() { searchTimer = null; refreshOptions_(query); }, 200); }); // mousedown fires before blur so selectedIndex is already updated; setTimeout(0) // lets the browser finish updating the selection before we read it rootSelect.addEventListener('mousedown', function() { setTimeout(function() { commitSelection_(); }, 0); }); // change as fallback (e.g. keyboard selection) rootSelect.addEventListener('change', function() { commitSelection_(); }); return { clear: function() { cancelSearch_(); rootSelect.value = ''; updateDisplay_(); hidePanel_(); const evt = doc.createEvent('Event'); evt.initEvent('change', true, true); rootSelect.dispatchEvent(evt); }, open: openDropdown_, }; }
